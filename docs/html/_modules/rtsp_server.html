    def __init__(self, fps, img_shape, properties={}):
        super(SensorFactory, self).__init__(**properties)
        self.height = int(img_shape[1])
        self.width = int(img_shape[0])

        self.work = ""

        self.launch_string = 'appsrc name=source is-live=true block=true format=GST_FORMAT_TIME ' + caps_str + \
                         '! videoconvert ' + \
                         '! video/x-raw,format=I420 ' + \
                         '! x264enc tune=zerolatency speed-preset=fast ' + \
                         '! rtph264pay config-interval=1 pt=96 name=pay0 ' + \
                         '' # <<<=== add more configs here
        logging.info(self.launch_string)

[документация]    def on_need_data(self, src, length):
            frame = self.image.copy()
            if frame.shape[:2] != (self.height, self.width):
                frame = cv2.resize(frame, (self.width, self.height))
            data = frame.tobytes()
            buf = Gst.Buffer.new_allocate(None, len(data), None)
            buf.fill(0, data)
            buf.duration = self.duration
            timestamp = self.number_frames * self.duration
            buf.pts = buf.dts = int(timestamp)
            buf.offset = timestamp
            self.number_frames += 1
            retval = self.appsrc.emit('push-buffer', buf)
            if retval != Gst.FlowReturn.OK:
                    logging.info("retval not OK: {}".format(retval))

[документация]class GstServer(GstRtspServer.RTSPServer):
    def __init__(self, fps:int, resolution:list,port:int,pt_mount:str, properties={}):
        Gst.init(None)
        super(GstServer, self).__init__(**properties)
        self.set_service(str(port))
        self.factory = SensorFactory(fps=fps, img_shape=resolution)
        self.factory.set_shared(True)
        self.get_mount_points().add_factory('/'+pt_mount, self.factory)
        self.attach(None)
        self.loop = None

        self.loop = GLib.MainLoop()

        t2.start()
