                    valid_res = [r for r in rc[0]] #if check_text_pattern(res[1][0])]

                        scores = [r[1][1] for r in valid_res]

# сохрпнение изображения номера если стоит параметр

[документация]    def check_text_pattern(self,text):
        # Паттерн для текста
        pattern = r'^[A-Za-z]\d{3}[A-Za-z]{2}\d{2,3}$'
        t_text = None

        def replace_digits_and_letters(text):
            result = list(text)

            if len(result) > 5:            
                # Меняем символы на местах букв
                if result[0] == '0':
                    result[0] = 'O'
                elif result[0] == '8':
                    result[0] = 'B'
                elif result[0] == 'V' or result[0] == 'v':
                    result[0] = 'Y'
                               
                if result[4] == '0':
                    result[4] = 'O'
                elif result[4] == '8':
                    result[4] = 'B'
                elif result[4] == 'V' or result[4] == 'v':
                    result[4] = 'Y'
                
                if result[5] == '0':
                    result[5] = 'O'
                elif result[5] == '8':
                    result[5] = 'B'
                elif result[5] == 'V' or result[5] == 'v':
                    result[5] = 'Y'
                
                # Меняем символы на местах цифр
                if result[1] == 'O':
                    result[1] = '0'
                elif result[1] == 'B':
                    result[1] = '8'
                
                if result[2] == 'O':
                    result[2] = '0'
                elif result[2] == 'B':
                    result[2] = '8'
                
                if result[3] == 'O':
                    result[3] = '0'
                elif result[3] == 'B':
                    result[3] = '8'
                
                                
                # Цифровые позиции могут продолжаться до конца строки в зависимости от паттерна
                for i in range(6, len(result)):
                    if result[i] == 'O':
                        result[i] = '0'
                    elif result[i] == 'B':
                        result[i] = '8'
            
            return ''.join(result)
        # Преобразуем текст по правилам замены
        transformed_text = replace_digits_and_letters(text)     
        # Проверяем соответствие паттерну
        if bool(re.match(pattern, transformed_text)):
            t_text = transformed_text.upper()  #Преобразуем текст в верхний регистр 
        else: t_text = None
        return t_text

        """ Увеличение качества изображения или фильтрация перед распознаванием
        номера OCR"""
        img = image.copy()
        img = cv2.GaussianBlur(img, (3, 3), 0) 
        return img 


[документация]    def qualityUp1(self,image):
        """ Увеличение качества изображения или фильтрация перед распознаванием
        номера OCR"""
        img = image.copy()
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE,(3,3))
        img = cv2.morphologyEx(img, cv2.MORPH_OPEN, kernel)
        img = cv2.GaussianBlur(img, (3, 3), 0) 
        return img
