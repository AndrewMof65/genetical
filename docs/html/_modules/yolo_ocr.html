[документация]    def detect_num(self,images):
        """Распознавание номера - модель YOLO_OBB
        Свой датасет - обучение""" 
        imgs = []
        results = []
        res_map=[False]*len(images)
        for i,img in enumerate(images):
            if(img is not None):
                res_map[i]=True
                img = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
                img = cv2.cvtColor(img, cv2.COLOR_GRAY2RGB)
                if(i<2):
                    if(self.quality_up):img = self.qualityUp(img)
                    rc = self.ocr1_model.predict(source=img,verbose=False)[0]
                    results.append(rc)
                else:
                    rc = self.ocr2_model.predict(source=img,verbose=False)[0]
                    results.append(rc)
                imgs.append(img)
        if(len(imgs)==0):return [None]*len(images)
        plate_nums = self.yolo_proc(results,res_map,imgs)
# сохрпнение изображения номера если стоит параметр
        for k in range(len(res_map)):
            if(plate_nums[k] is not None and OCR_SAVE):
                    plt.imsave("ocr/box_"+str(k)+"_"+str( plate_nums[k])+"_"+str(self.img_num[k])+".jpg",images[k])
                    self.img_num[k]+=1
        if(DEBUG):print(plate_nums)
        return plate_nums    


[документация]    def yolo_proc(self,results,res_map,imgs):
        plate_nums = [None]*len(res_map)
        if(results is not None):
            i=0
            for k in range(len(res_map)):
                if(res_map[k]):
                    result = json.loads(results[i].to_json())
                    size=imgs[i].shape[1]
                    i+=1
                    confid = [data["confidence"] for data in result]
                    nums = [data["class"] for data in result]
                    xes=[data["box"]["x1"] for data in result]
                    zp = zip(xes,nums,confid)
                    zps = sorted(zp, key=lambda tup: tup[0])
                    sz=len(zps)
                    n = 0
                    while(True):
                        if (n+1>=sz):break
                        beg = int(zps[n][0])
                        end = int(zps[n+1][0])
                        if ( beg in range(end-5,end+5) ):
                            if(zps[n+1][2]>zps[n][2]):
                                del zps[n]
                            else: 
                                del zps[n+1]
                                n+=1
                            sz=len(zps)
                            continue
                        n+=1
                    # if(len(zps)>0 and zps[-1][2]<0.80):del zps[-1]
                    # if(len(zps)>0 and zps[0][2]<=0.80):del zps[0]
                    if (len(zps)>0 and size-zps[-1][0]<size/11): del zps[-1]
                    if (len(zps)>0 and zps[0][0]<5): del zps[0]
                    plate_nums[k] = ("".join( [self.letters[n[1]] for n in zps]))
                    if(plate_nums[k] != ""):
                        diff = re.fullmatch(r'^[A-Za-z]\d{3}[A-Za-z]{2}\d{2,3}$', plate_nums[k])
                        if diff is None:
                            diff = re.fullmatch(r'^[A-Za-z]\d{3}[A-Za-z]{2}\d{4}$', plate_nums[k])
                            if(diff is None):
                                plate_nums[k] = None
                            elif(list(plate_nums[k])[6]=="1"):
                                    lst=list(plate_nums[k])
                                    del(lst[6])
                                    plate_nums[k]= ''.join(lst)
                            else: plate_nums[k] = None
                    else: plate_nums[k] = None 
                else: continue
        return plate_nums


[документация]    def qualityUp(self,image):
        """ Увеличение качества изображения или фильтрация перед распознаванием
        номера OCR"""
        img = image.copy()
        mean = np.mean(img)
        img[img>mean]=255
        img[img<mean]=0
        img = cv2.GaussianBlur(img,(3,3),3)
        kernel = np.ones((2, 2), np.uint8) 
        img = cv2.erode(img, kernel,iterations=2) 
        img = cv2.dilate(img, kernel,iterations=2) 
# ****** Черную рамочку вокруг номера ***************************************
        wdth = 3
        img[-wdth:,:]=0
        img[0:wdth,:]=0
        img[:,-wdth:]=0
        img[:,0:wdth]=0
        if(DEBUG):
            print("QualityUp")
            plt.imshow(img)
            plt.show()
        return img
