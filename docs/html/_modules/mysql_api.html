[документация]    def mysql_connect(self):
        """ Функция соединения с базой данных"""
        try:
            mydb = connect(
                host=self.bd_host,
                port = self.bd_port,
                user=self.bd_login,
                password=self.bd_passwd,
                database=self.bd_db
            )
            create_tbl_query = """DROP TABLE IF EXISTS detector;
                                 CREATE TABLE detector(
                                 id INT AUTO_INCREMENT PRIMARY KEY,
                                 room VARCHAR(255) NOT NULL DEFAULT '',
                                 box_num INT NOT NULL DEFAULT 0,
                                 is_car_present TINYINT NOT NULL DEFAULT 0,
                                 car_num VARCHAR(16) NOT NULL DEFAULT '',
                                 mod_date DATETIME NOT NULL,
                                 CONSTRAINT `place` UNIQUE (room,box_num ))ENGINE="""
            create_tbl_query += self.bd_type +";"
            tbl_query = "INSERT INTO detector (room,box_num,is_car_present,mod_date)"+ \
                                     "VALUES(%s,%s,%s, TIMESTAMP(CURTIME()));"
            val=[]
            for i in range(self.box_count):
                val.append( (self.room_name,str(i+1), str(0)) )
            with mydb.cursor() as cursor:
                    cursor.execute(create_tbl_query)
            mydb.connect()
            with mydb.cursor() as cursor:
                    cursor.executemany(tbl_query,val)
                    mydb.commit()
            self.status = True                
            self.error = "Mysql connecting OK"
            return mydb
        except Error as e:
            self.error = "Mysql: {}".format(e)
            self.status = False
            logging.error("Mysql: {}".format(e),exc_info=True)
        return None
