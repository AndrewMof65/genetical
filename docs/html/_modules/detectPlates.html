        """Распознавание номера - модель YOLO_OBB
        Свой датасет - обучение"""
        size = 520
        imgs = []
        res=[None]*len(images)
        plate_nums = [None]*len(images)
        for i,img in enumerate(images):
            if(img is not None):
                res[i]=0
                if(self.quality_up):img = self.qualityUp(img)
                img=cv2.resize(img,(size,int(img.shape[0]/img.shape[1] * size)), 
                                       interpolation= cv2.INTER_LINEAR)
                img = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
                img = cv2.cvtColor(img, cv2.COLOR_GRAY2RGB)
                imgs.append(img)
            else:res[i]=None
        if(len(imgs)==0):return plate_nums
        results = self.ocr_model.predict(source=imgs,verbose=False)
        if(results is not None):
            i=0
            for k in range(len(res)):
                if(res[k] is not None):
                    result = json.loads(results[i].tojson())
                    i+=1
                    confid = [data["confidence"] for data in result]
                    nums = [data["class"] for data in result]
                    xes=[data["box"]["x1"] for data in result]
                    zp = zip(xes,nums,confid)
                    zps = sorted(zp, key=lambda tup: tup[0])
                    sz=len(zps)
                    n = 0
                    while(True):
                        if (n+1>=sz):break
                        beg = int(zps[n][0])
                        end = int(zps[n+1][0])
                        if ( beg in range(end-5,end+5) ):
                            if(zps[n+1][2]>zps[n][2]):
                                del zps[n]
                            else: 
                                del zps[n+1]
                                n+=1
                            sz=len(zps)
                            continue
                        n+=1
                    if(len(zps)>0 and zps[-1][2]<0.80):del zps[-1]
                    if(len(zps)>0 and zps[0][2]<=0.80):del zps[0]
                    if (len(zps)>0 and size-zps[-1][0]<50): del zps[-1]
                    if (len(zps)>0 and zps[0][0]<10): del zps[0]
                    plate_nums[k] = ("".join( [self.letters[n[1]] for n in zps]))
                    if(plate_nums[k] != ""):
                        diff = re.fullmatch(r'^[A-Za-z]\d{3}[A-Za-z]{2}\d{2,3}$', plate_nums[k])
                        if diff is None:
                            plate_nums[k] = None 
                    else: plate_nums[k] = None 
                else: continue
        return plate_nums


[документация]    def yolo_obb(self,images):
        """ Метод распознавания рамок от YOLO_OBB 
         Свой датасет"""
         
        size = 800
        imgs =[]
        coef = []
        for old_im in images:
            if(old_im is not None):
                coef.append(size/old_im.shape[1])
                old_im = cv2.resize(old_im, 
                                    (size,int(old_im.shape[0]/old_im.shape[1] * size)), 
                                    interpolation= cv2.INTER_LINEAR)
                image = old_im.copy()        
                imgs.append(image)
        out_data = [[None,None]]*len(images)
        if(len(imgs)==0): return out_data
        results = self.obb_model.predict(source=imgs,verbose=False)
        if(results is not None): 
                cnt=0
                for i,result in enumerate(results):
                    while(images[cnt] is None):cnt+=1
                    xywhr = [res.obb.xywhr for res in result]
                    if(len(xywhr)>0):
                        img = imgs[i] # [res.orig_img for res in result][0]
                        dt = xywhr[0][0] # считаем только один bounding box на каждый кадр
                        rect = [(float(dt[0]),float(dt[1])),
                                  (float(dt[2]),float(dt[3])),float(dt[4])]
                        rect[2]= np.rad2deg(rect[2])
                        if(rect[2]>90):rect[2]=rect[2]-180
                        [cx,cy,w,h,angle] = [rect[0][0],rect[0][1],rect[1][0],rect[1][1],rect[2]]
                        im_out = self.rotate_image(img, angle,center = (int(cx),int(cy)))
                        x=cx-w/2
                        y=cy-h/2
                        if(x<0):x=0
                        if(y>img.shape[0]):y=img.shape[0]
                        im_out = im_out[ int(y):int(y+h),int(x):int(x+w)]
                        box = cv2.boxPoints(rect) # поиск четырех вершин прямоугольника
                        box = np.intp(box) 
                        out_data[cnt] = [im_out,(box/coef[i]).astype(int)]
                        if(ANNOTATION):self.saveInDataSet(img,box,cnt)
                        if(DEBUG):
                            plt.imshow(im_out)
                            plt.show()
                            print("Прямоугольник",rect)
                            cv2.drawContours(img,[box],0,(255,0,0),3) # рисуем прямоугольник
                            plt.imshow(img)
                            plt.show()
                    cnt+=1
        return out_data

        """ Осветление изображения"""
        hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
        h, s, v = cv2.split(hsv)
        lim = 255 - value
        v[v > lim] = 255
        v[v <= lim] += value
        final_hsv = cv2.merge((h, s, v))
        img = cv2.cvtColor(final_hsv, cv2.COLOR_HSV2BGR)
        return img

        """ Осветление/затемнение изображения"""
        hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
        h, s, v = cv2.split(hsv)
        v = cv2.add(v,value)
        v[v > 255] = 255
        v[v < 0] = 0
        final_hsv = cv2.merge((h, s, v))
        img = cv2.cvtColor(final_hsv, cv2.COLOR_HSV2BGR)
        return img

[документация]    def rotate_image(self,image, angle,center = None):
        """ Поворот изображения вокруг центра center на угол angle"""
        if(center is None):image_center = tuple(np.array(image.shape[1::-1]) / 2)
        else: image_center = center #tuple(np.array(image.shape[1::-1]) / 2)
        rot_mat = cv2.getRotationMatrix2D(image_center, angle, 1.0)
        result = cv2.warpAffine(image, rot_mat, image.shape[1::-1], flags=cv2.INTER_LINEAR)
        return result

        img = image.copy()

            print("QualityUp")

[документация]    def imgs_of_boxes(self,full_image,shots,allboxes,boxnum):
        """Разделение кадра на 4 бокса для авто"""
        if(allboxes):
            box_size = len(self.boxes)
            boxes = range(box_size)
        else: 
            box_size = 1
            boxes = [boxnum]
        images = [None] * box_size
        for i in boxes: # выделение боксов на картинке и увеличение из разрешения
                cv2.rectangle(full_image, **self.boxes[i])
                x1,y1 = self.boxes[i]["pt1"][0],self.boxes[i]["pt1"][1]
                x2,y2 = self.boxes[i]["pt2"][0],self.boxes[i]["pt2"][1]
                up_width = int((x2-x1)*self.k_size[i]) # увеличение разрешения
                up_height = int((y2-y1)*self.k_size[i])
                images[i] = cv2.resize(shots[y1:y2, x1:x2], (up_width,up_height), interpolation= cv2.INTER_LINEAR)    
        return images

        success=True  
        if(not allboxes): self.boxes = [self.boxes[boxnum]]

        nums_pool = [None] * box_size
        for i in range(box_size): nums_pool[i]=[]
        plate_nums = [None] * box_size
        frame_sum = self.one_in      
        inverse=1/self.one_in
        shots = None
        self.rtsp_server.run()
        while(success):
            self.cap = cv2.VideoCapture(self.capture_url, cv2.CAP_FFMPEG)
            self.cap.set(cv2.CAP_PROP_BUFFERSIZE, 0)
            if not  self.cap.isOpened():
                logging.info('Cannot open RTSP stream: {}'.format(self.capture_url))
                continue
            success, img = self.cap.read()
            self.cap.release()
            if(not success):break
            img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)           
            if(img.shape != self.resolution):
                img = cv2.resize(img, self.resolution, interpolation= cv2.INTER_LINEAR)
            nimg = np.zeros(img.shape)
            img = cv2.normalize(img,  nimg, 0, 255, cv2.NORM_MINMAX)
            transparent = np.full(img.shape,255, dtype=np.uint8)
            if(frame_sum == 0):frame_sum = self.one_in
            if(frame_sum < self.one_in and (shots is not None)): 
                shots += cv2.addWeighted(transparent, 1-inverse, img,inverse, 0)
                frame_sum -=1
            else:
                if(self.one_in == 1):  shots = img.copy()
                else: shots = cv2.addWeighted(transparent, 1-inverse, img, inverse, 0)
                frame_sum -=1
            if(DEBUG):
                if(self.cnt_cadr == NUMshots ):
                    frame_sum = self.one_in ;
                    self.cnt_cadr+=1
                    print("\rКадры: ",self.cnt_cadr,end="                  ")
                    continue;
                if(self.cnt_cadr < NUMshots ): 
                    print("\rКадры: ",self.cnt_cadr,end="                  ")
                    self.cnt_cadr+=1
                    continue
                if self.cnt_cadr > NUMshots +self.one_in: break
            print("\rКадры: ",self.cnt_cadr,end="                   ")
            self.cnt_cadr+=1
            box_images = self.imgs_of_boxes(img,shots,allboxes,boxnum)
            tmp_img = img.copy()
            cv2.putText(tmp_img, "Readen frame ", (980,img.shape[0]-200), cv2.FONT_HERSHEY_COMPLEX_SMALL, 2, (255,0,0), 2)
            tmp = cv2.cvtColor(tmp_img.copy(), cv2.COLOR_RGB2BGR)
            self.rtsp_server.factory.set_image(tmp)
            if(frame_sum == 0):  
                box_images = self.cars_present(box_images) 
                text_in="Car in boxes: "
                text_out="Car out boxes: "
                for i,im in enumerate(box_images):
                    if(im is not None):
                        text_in+=str(i)+" "
                        cv2.putText(tmp_img, text_in, (980,img.shape[0]-100), cv2.FONT_HERSHEY_COMPLEX_SMALL, 2, (255,0,0), 2)
                        # +++++++Запись в базу - машина в боксе
                        tbl_query = " UPDATE detector SET is_car_present=%s WHERE room=%s AND box_num=%s;"
                        val=('1',self.room,str(i+1))
                        self.mysql_query(tbl_query,val)
                        logger.debug('\nВ боксе {} машина'.format(str(i)))

                    else:
                        text_out+=str(i)+" "
                        cv2.putText(tmp_img, text_out, (980,img.shape[0]-50), cv2.FONT_HERSHEY_COMPLEX_SMALL, 2, (255,0,0), 2)
                        # +++++++Запись в базу - бокс пустой
                        plate_nums[i] = None
                        tbl_query = " UPDATE detector SET is_car_present=%s,car_num='' WHERE room=%s AND box_num=%s;"

                        self.mysql_query(tbl_query,val)
