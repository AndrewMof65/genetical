[документация]    def check_text_pattern(self,text):
        # Паттерн для текста
        pattern = r'^[A-Za-z]\d{3}[A-Za-z]{2}\d{2,3}$'
        t_text = None

        def replace_digits_and_letters(text):
            result = list(text)

            if len(result) > 5:            
                # Меняем символы на местах букв
                if result[0] == '0':
                    result[0] = 'O'
                elif result[0] == '8':
                    result[0] = 'B'
                elif result[0] == 'V' or result[0] == 'v':
                    result[0] = 'Y'
                               
                if result[4] == '0':
                    result[4] = 'O'
                elif result[4] == '8':
                    result[4] = 'B'
                elif result[4] == 'V' or result[4] == 'v':
                    result[4] = 'Y'
                
                if result[5] == '0':
                    result[5] = 'O'
                elif result[5] == '8':
                    result[5] = 'B'
                elif result[5] == 'V' or result[5] == 'v':
                    result[5] = 'Y'
                
                # Меняем символы на местах цифр
                if result[1] == 'O':
                    result[1] = '0'
                elif result[1] == 'B':
                    result[1] = '8'
                
                if result[2] == 'O':
                    result[2] = '0'
                elif result[2] == 'B':
                    result[2] = '8'
                
                if result[3] == 'O':
                    result[3] = '0'
                elif result[3] == 'B':
                    result[3] = '8'
                
                                
                # Цифровые позиции могут продолжаться до конца строки в зависимости от паттерна
                for i in range(6, len(result)):
                    if result[i] == 'O':
                        result[i] = '0'
                    elif result[i] == 'B':
                        result[i] = '8'
            
            return ''.join(result)
        # Преобразуем текст по правилам замены
        transformed_text = replace_digits_and_letters(text)     
        # Проверяем соответствие паттерну
        if bool(re.match(pattern, transformed_text)):
            t_text = transformed_text.upper()  #Преобразуем текст в верхний регистр 
        else: t_text = None
        return t_text

[документация]    def yolo_proc(self,results,res_map,imgs):
        plate_nums = [None]*len(res_map)
        if(results is not None):
            i=0
            for k in range(len(res_map)):
                if(res_map[k]):
                    result = json.loads(results[i].to_json())
                    size=imgs[i].shape[1]
                    i+=1
                    confid = [data["confidence"] for data in result]
                    nums = [data["class"] for data in result]
                    xes=[data["box"]["x1"] for data in result]
                    zp = zip(xes,nums,confid)
                    zps = sorted(zp, key=lambda tup: tup[0])
                    sz=len(zps)
                    n = 0
                    while(True):
                        if (n+1>=sz):break
                        beg = int(zps[n][0])
                        end = int(zps[n+1][0])
                        if ( beg in range(end-5,end+5) ):
                            if(zps[n+1][2]>zps[n][2]):
                                del zps[n]
                            else: 
                                del zps[n+1]
                                n+=1
                            sz=len(zps)
                            continue
                        n+=1
                    # if(len(zps)>0 and zps[-1][2]<0.80):del zps[-1]
                    # if(len(zps)>0 and zps[0][2]<=0.80):del zps[0]
                    if (len(zps)>0 and size-zps[-1][0]<size/11): del zps[-1]
                    if (len(zps)>0 and zps[0][0]<5): del zps[0]
                    plate_nums[k] = ("".join( [self.letters[n[1]] for n in zps]))
                    if(plate_nums[k] != ""):
                        diff = re.fullmatch(r'^[A-Za-z]\d{3}[A-Za-z]{2}\d{2,3}$', plate_nums[k])
                        if diff is None:
                            diff = re.fullmatch(r'^[A-Za-z]\d{3}[A-Za-z]{2}\d{4}$', plate_nums[k])
                            if(diff is None):
                                plate_nums[k] = None
                            elif(list(plate_nums[k])[6]=="1"):
                                    lst=list(plate_nums[k])
                                    del(lst[6])
                                    plate_nums[k]= ''.join(lst)
                            else: plate_nums[k] = None
                    else: plate_nums[k] = None 
                else: continue
        return plate_nums

        """ Осветление изображения"""
        hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
        h, s, v = cv2.split(hsv)
        lim = 255 - value
        v[v > lim] = 255
        v[v <= lim] += value
        final_hsv = cv2.merge((h, s, v))
        img = cv2.cvtColor(final_hsv, cv2.COLOR_HSV2BGR)
        return img


[документация]    def change_brightness(self,img, value=30):
        """ Осветление/затемнение изображения"""
        hsv = cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
        h, s, v = cv2.split(hsv)
        v = cv2.add(v,value)
        v[v > 255] = 255
        v[v < 0] = 0
        final_hsv = cv2.merge((h, s, v))
        img = cv2.cvtColor(final_hsv, cv2.COLOR_HSV2BGR)
        return img

[документация]    def rotate_image(self,image, angle,center = None):
        """ Поворот изображения вокруг центра center на угол angle"""
        if(center is None):image_center = tuple(np.array(image.shape[1::-1]) / 2)
        else: image_center = center #tuple(np.array(image.shape[1::-1]) / 2)
        rot_mat = cv2.getRotationMatrix2D(image_center, angle, 1.0)
        result = cv2.warpAffine(image, rot_mat, image.shape[1::-1], flags=cv2.INTER_LINEAR)
        return result

    

[документация]    def qualityUp(self,image):
        """ Увеличение качества изображения или фильтрация перед распознаванием
        номера OCR"""
        img = image.copy()
        mean = np.mean(img)
        img[img>mean]=255
        img[img<mean]=0
        img = cv2.GaussianBlur(img,(3,3),3)
        kernel = np.ones((2, 2), np.uint8) 
        img = cv2.erode(img, kernel,iterations=2) 
        img = cv2.dilate(img, kernel,iterations=2) 
# ****** Черную рамочку вокруг номера ***************************************
        wdth = 3
        img[-wdth:,:]=0
        img[0:wdth,:]=0
        img[:,-wdth:]=0
        img[:,0:wdth]=0
        if(DEBUG):
            print("QualityUp")
            plt.imshow(img)
            plt.show()
        return img

                    tbl_query = " UPDATE detector SET is_car_present=%s WHERE room=%s AND box_num=%s;"

                    # +++++++Запись в базу - бокс пустой

                    tbl_query = " UPDATE detector SET is_car_present=%s,car_num='' WHERE room=%s AND box_num=%s;"
                    val=('0',self.room,str(i+1))
                    self.mysql_query(tbl_query,val)
                    logger.debug('\nВ боксе {} машины НЕТ'.format(str(i))) 
            nums = self.main_ocr([d[0] for d in self.data_out])  
            for i in range(box_size):
                if(nums[i] is not None): 
                    if(DEBUG):self.plate_nums[i] = nums[i] 
                    self.nums_pool[i].append(nums[i])
                    if(len(self.nums_pool[i])==self.poolsize):
                        D = defaultdict(list)
                        for idx,item in enumerate(self.nums_pool[i]):
                            D[item].append(idx)
                        D = {k:v for k,v in D.items()}
                        conf=0
                        for num,idxs in D.items():
                            if(len(idxs)>conf):
                                conf=len(idxs)
                                self.plate_nums[i] = num  
                        self.nums_pool[i].clear()
                # +++++++Запись в базу номера машины для i-го бокса plate_nums[i]
                        tbl_query = " UPDATE detector SET car_num=%s WHERE room=%s AND box_num=%s;"
                        val=(self.plate_nums[i],self.room,str(i+1))
                        self.mysql_query(tbl_query,val)
                else:  self.nums_pool[i].clear()
            if (DEBUG):break
