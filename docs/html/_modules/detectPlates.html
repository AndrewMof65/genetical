[документация]    def imgs_of_boxes(self,full_image,allboxes,boxnum):
        """Разделение кадра на 4 бокса для детектирования авто и их номеров"""
        if(allboxes):
            box_size = len(self.boxes)
            boxes = range(box_size)
        else: 
            box_size = 1
            boxes = [boxnum]
        images = [None] * box_size
        for i in boxes: 
                cv2.rectangle(full_image, **self.boxes[i])
                """выделение боксов на картинке"""
                x1,y1 = self.boxes[i]["pt1"][0],self.boxes[i]["pt1"][1]
                x2,y2 = self.boxes[i]["pt2"][0],self.boxes[i]["pt2"][1]
                up_width = int((x2-x1)*self.k_size[i]) # увеличение разрешения
                up_height = int((y2-y1)*self.k_size[i])
                images[i] = cv2.resize(full_image[y1:y2, x1:x2], (up_width,up_height), interpolation= cv2.INTER_LINEAR)    
                """Установка оптимального разрешения области бокса для детекции"""
        return images

            

[документация]    def draw_out_img(self,img,data_out,plate_nums): 
            """Функция визуализации детектированных данных"""
            box_size = len(self.boxes)
            imgRoi = [None] * box_size
            plate = [None] * box_size
            bx = [None] * box_size
            for i in range(box_size):                  
                if(len(data_out)>0):
                    [recogn,box] = data_out[i]
                else: [recogn,box] = [None,None]
                x1,y1 = self.boxes[i]["pt1"][0],self.boxes[i]["pt1"][1]
                x2,y2 = self.boxes[i]["pt2"][0],self.boxes[i]["pt2"][1]
# ****************************** Выравнивание рамки и распознавание номера ***********************
                if( recogn is not None):
# ****************************** Разметка подгон изображения рамки *********************** 
                    bx[0] = [int(x1+box[0][0]/self.k_size[i]),int(y1+box[0][1]/self.k_size[i])]
                    bx[1] = [int(x1+box[1][0]/self.k_size[i]),int(y1+box[1][1]/self.k_size[i])]
                    bx[2] = [int(x1+box[2][0]/self.k_size[i]),int(y1+box[2][1]/self.k_size[i])]
                    bx[3] = [int(x1+box[3][0]/self.k_size[i]),int(y1+box[3][1]/self.k_size[i])]
                    plate[i] = np.array(bx)
                    imgRoi[i] = cv2.resize(recogn, (int(recogn.shape[1]*100/recogn.shape[0]),100), interpolation= cv2.INTER_LINEAR)
# ******************** Напишем распознаный номер на общем изображении *******************************************************                    
                if( plate_nums[i] is not None):
                    cv2.putText(img, " "+plate_nums[i], (self.boxes[i]["pt1"][0],self.boxes[i]["pt1"][1]+50), cv2.FONT_HERSHEY_COMPLEX_SMALL, 2, (255,0,0), 2)
                    logger.debug("Номер машины",plate_nums[i])
                if( plate[i] is not None ):         
 # ****************** Наложим изображение рамки на общую картинку *****************************
                    x_offset = x1
                    y_offset = 5 
                    if(i==1 or i==2):x_offset = x1 - 100
                    elif(i==3): x_offset = x1 + 50
                    if(x_offset + imgRoi[i].shape[1]>img.shape[1]):
                        d= x_offset + imgRoi[i].shape[1]-img.shape[1]
                        imgRoi[i] = cv2.resize(imgRoi[i], (imgRoi[i].shape[1]-d,imgRoi[i].shape[0]), interpolation= cv2.INTER_LINEAR)
                    img[y_offset:y_offset+ imgRoi[i].shape[0], x_offset:x_offset+ imgRoi[i].shape[1]] = imgRoi[i]
# ******************** Обведем рамкой  номер на общем изображении ***********************************
                    cv2.drawContours(img,[plate[i]],0,(255,0,0),2)
            return img

            print(error)
            cv2.putText(img, error, (50,int(img.shape[0]/2)), cv2.FONT_ITALIC, 2, (255,0,0),3)
        if(len(info)>0 and len(info)<4 ):

                print(txt)
                cv2.putText(img, txt, (980,img.shape[0]-pos_y[i]), cv2.FONT_HERSHEY_COMPLEX_SMALL, 2, (255,0,0),2)

        return img



[документация]    def detection(self):
        """Фуекция детектирования авто в боксах, их номерных рамок и определение самих номнров"""
        while(True):
            box_images = self.box_images 
            if(self.box_images is None):
                time.sleep(1)
                continue
            box_size = len(self.boxes)
            self.data_out = self.yolo_cars.yolo_obb(box_images)
            imgs = self.yolo_cars.cars_present(box_images.copy()) 
            for i,dt in enumerate(self.data_out):
                if(dt[0] is not None and imgs[i] is None):imgs[i] = box_images[i].copy() 
            self.text_in="Car in boxes: "
            self.text_out="Car out boxes: "
            for i,im in enumerate(imgs):
                if(im is not None):
                    self.text_in+=str(i+1)+" "
                    self.update_tbl("1", "", self.room, str(i+1))
                    """Запись в базу статуса - машина в боксе"""
                    logger.debug('\nВ боксе {} машина'.format(str(i))) 
                else:
                    self.text_out+=str(i+1)+" "
                    self.plate_nums[i] = None
                    self.update_tbl("0", "", self.room, str(i+1))
                    """Запись в базу статуса - бокс пустой"""
                    logger.debug('\nВ боксе {} машины НЕТ'.format(str(i))) 
            nums = self.ocr.detect_num([d[0] for d in self.data_out])  
            """Определение номеров автомобилей в боксе"""
            for i in range(box_size):
                if(nums[i] is not None): 
                    if(DEBUG):self.plate_nums[i] = nums[i] 
                    self.nums_pool[i].append(nums[i])
                    if(len(self.nums_pool[i])==self.poolsize):
                        D = defaultdict(list)
                        for idx,item in enumerate(self.nums_pool[i]):
                            """Алгоритм детектирвания номера много раз и выборка наиболее частого значения"""
                            D[item].append(idx)
                        D = {k:v for k,v in D.items()}
                        conf=0
                        for num,idxs in D.items():
                            if(len(idxs)>conf):
                                conf=len(idxs)
                                self.plate_nums[i] = num  
                        self.nums_pool[i].clear()
                        self.update_tbl("1", self.plate_nums[i], self.room, str(i+1))
                        """Запись в базу номера машины для i-го бокса plate_nums[i]"""
                else:  self.nums_pool[i].clear()
            if (DEBUG):break

    

[документация]    def run(self,poolsize=1,allboxes=False,boxnum=0,test=""):
        """Основная функция с бесконечным циклом считывания кадров и детектированием рамок и распознавания номеров"""
        success=True  
        self.poolsize = poolsize
        self.cnt_cadr = 0
        if(not allboxes): self.boxes = [self.boxes[boxnum]]
        box_size = len(self.boxes)
        self.nums_pool = [None] * box_size
        for i in range(box_size): self.nums_pool[i]=[]
        self.plate_nums = [None] * box_size
        self.data_out = [[None,None]]*box_size 
        self.text_in=""
        self.text_out=""
        img_control = self.insert_img_control(None,info = ["START!!"])
        if(DEBUG or TEST): 
            self.cap = cv2.VideoCapture("resourses/"+test+".mp4")
            if(TEST):
                fourcc = cv2.VideoWriter_fourcc(*'mp4v')
                self.out = cv2.VideoWriter("resourses/"+test+"_out.mp4", fourcc, 30.0, (1920,1080))
        if(not DEBUG):
            t2 = threading.Thread(target=self.detection, daemon=False)
            t2.start() 
        while(success):
            if(not (DEBUG or TEST)): 
                self.cap = cv2.VideoCapture(self.capture_url, cv2.CAP_FFMPEG)
                self.cap.set(cv2.CAP_PROP_BUFFERSIZE, 0)
            if not  self.cap.isOpened():
                if(TEST):print("resourses/"+test+"_out.mp4")
                logmsg='Cannot open RTSP stream: {}'.format(self.capture_url)
                logging.info(logmsg)
                img_control = self.insert_img_control(None,error=logmsg)
                time.sleep(1)
                continue
            else: 
                if(not TEST): print('Open RTSP stream: {}  - OK'.format(self.capture_url))
            success, img = self.cap.read()
            if(not (DEBUG or TEST)):self.cap.release()
            if(not success): 
                if(TEST):continue
                img_control = self.insert_img_control(None,error="Image read error")
                success=True
                time.sleep(1)
                continue
            img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
            if([img.shape[1],img.shape[0]] != [self.resolution[0],self.resolution[1]]):
                img = cv2.resize(img, self.resolution, interpolation= cv2.INTER_LINEAR)
            print("\rКадры: ",self.cnt_cadr,end="                   ")
            if(DEBUG):
                if(self.cnt_cadr < NUMshots ): 
                    self.cnt_cadr+=1
                    continue
                if self.cnt_cadr > NUMshots: break
            self.cnt_cadr+=1
            self.box_images = self.imgs_of_boxes(img,allboxes,boxnum)
            if(DEBUG): self.detection()
            img_control = self.draw_out_img(img.copy(),self.data_out,self.plate_nums)
            img_control = self.insert_img_control(
                img_control,info=["Detect numbers",self.text_in,self.text_out])
            if(TEST): 
                self.out.write(cv2.cvtColor(img_control, cv2.COLOR_RGB2BGR))
            else: time.sleep(1)
            if(DEBUG):       
                plt.imshow(img_control)
                plt.show()
        if(TEST): 
            self.out.release()
            self.cap.release()
